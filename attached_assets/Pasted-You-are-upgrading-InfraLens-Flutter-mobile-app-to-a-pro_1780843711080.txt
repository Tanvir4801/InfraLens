            uptime_seconds, container_count}

fetchServers() → GET http://10.194.193.201:8000/api/servers
  Returns: [{name, ip, role, cpu, ram, disk,
             uptime, status: healthy/warning/down}]

fetchAlerts() → GET http://10.194.193.201:8000/api/alerts
  Returns: [{id, name, severity, description,
             fired_at, source}]

fetchPrediction() → GET http://10.194.193.201:8000/api/predict
  Returns: {will_overload, predicted_max_cpu,
            minutes_until_overload, confidence,
            data_points_used, forecast: [...]}

fetchServerHistory(name) →
  GET http://10.194.193.201:8000/api/servers/{name}/history
  Returns: [{timestamp, cpu, ram}]

postIncidentReport(alert) →
  POST http://10.194.193.201:8000/api/incident-report
  Body: {alert: {...}, metrics: {cpu_history, ram_history}}
  Returns: {report: "text...", model: "gemini-1.5-flash"}

postAiChat(question, metrics) →
  GET http://10.194.193.201:8000/api/chat?q={question}
  Returns: {response: "text..."}

acknowledgeAlert(id) →
  PATCH http://10.194.193.201:8000/api/alerts/{id}/acknowledge

restartContainer(id) →
  POST http://10.194.193.201:8000/api/containers/{id}/restart
  Returns: {success: bool, message: string}
